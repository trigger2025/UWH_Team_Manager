PASS 35 — FINAL SAFARI SQUASH FIX

Problem  
Mobile Safari scales off-screen elements to the viewport width before html2canvas captures them.  
This causes the exported image to appear vertically squashed.

Solution  
Force the export container to render at a fixed desktop width before capture.

--------------------------------------------------

1. Replace your export function with this

import html2canvas from "html2canvas";
import { createRoot } from "react-dom/client";
import TeamsExportLayout from "../components/export/TeamsExportLayout";

export async function exportTeams(teams, options) {

  const wrapper = document.createElement("div");

  wrapper.style.position = "fixed";
  wrapper.style.left = "-10000px";
  wrapper.style.top = "0";
  wrapper.style.width = "1000px";
  wrapper.style.transform = "scale(1)";
  wrapper.style.transformOrigin = "top left";
  wrapper.style.background = "#0f172a";

  document.body.appendChild(wrapper);

  const root = createRoot(wrapper);

  root.render(
    <TeamsExportLayout
      teams={teams}
      includeRatings={options.includeRatings}
      includePositions={options.includePositions}
    />
  );

  await new Promise(resolve => setTimeout(resolve, 200));

  const canvas = await html2canvas(wrapper, {
    scale: 3,
    useCORS: true,
    backgroundColor: "#0f172a",
    windowWidth: 1000,
    width: 1000
  });

  const image = canvas.toDataURL("image/png");

  document.body.removeChild(wrapper);

  const link = document.createElement("a");
  link.href = image;
  link.download = "generated-teams.png";
  link.click();
}

--------------------------------------------------

2. Ensure your export layout width is fixed

Inside TeamsExportLayout.jsx the outer container must include:

style={{
  width: "1000px",
  minWidth: "1000px",
  maxWidth: "1000px",
  padding: 40,
  background: "#0f172a",
  color: "white"
}}
